What is Lodash?

Lodash is a modern JavaScript utility library that makes JavaScript easier by taking the hassle out of working with arrays, numbers, objects, strings, etc.

Why do I need it?

You can interact with data types and structures by implementing your algorithms, but in some cases using Lodash will leverage your productivity and avoid some lines of code.

How do I use it?

							
								$ npm install lodash

After installation is complete, Lodash can be imported into a JavaScript file as:

							
								import _ from "lodash"

Let's get started

_.isEmpty

						
							// Empty array
							_.isEmpty([]) // true
							// Empty object
							_.isEmpty({}) // true
							// Empty string
							_.isEmpty('') // true
							// null
							_.isEmpty(null) // true
							// undefined
							_.isEmpty(undefined) // true

_.get

						
							const aryaStark = {
								name: 'Arya Stark',
								age: 15,
								father: { name: 'Ned Start' },
								mother: { name: 'Catelyn Stark' }
							}
							const jonSnow = {
								name: 'Jon Snow',
								age: 30,
								father: { name: 'Ned Stark' },
								mother: null
							}
							get('mother.name', aryaStark) // 'Catelyn Stark'
							get('mother.name', jonSnow) // undefined

_.get

						
							const characters = [
								{
									name: 'Jon Snow',
									house: 'Stark',
									parents: ['Ned Stark', 'Unknown Mom (spoiler)']
								},
								{
									name: 'Jaime Lannister',
									house: 'Lannister',
									parents: ['Tywin Lannister', 'Joanna Lannister']
								}
							];
							_.get('[1].parents[0]', characters); // Tywin Lannister

_.pick

						
							const jonSnow = {
								firstName: 'Jon',
								lastName: 'Snow',
								isBastard: true,
								father: {
									firstName: 'Ned',
									lastName: 'Stark'
								},
								mother: 'Unknown',
								age: 14,
								wolf: 'Ghost'
							}
							_.pick(['firstName', 'father.firstName', 'wolf'], jonSnow)
							// { firstName: 'Jon' father: { firstName: 'Ned' } wolf: 'Ghost' }

_.sample

Get a random item from a list

						
							const months = ['January', 'February', 'March', 'April', 'May',
							'June', 'July', 'August', 'September', 'October', 'November', 
							'December'];
							// Common variant
							function getRandomItem(list){
								return list[Math.floor(Math.random() * (list.length))];
							}
							getRandomItem(months);
							// Lodash
							_.sample(months);

_.intersection

Intersection of arrays

						
							const arrayA = [1, 2, 3, 4, 5];
							const arrayB = [2, 3];
							// Common variant
							function intersection(a, b) {
								return a.filter(Set.prototype.has, new Set(b));
							}
							intersection(arrayA, arrayB); // [2, 3]
							// Lodash
							_.intersection(arrayA, arrayB);

_.times

Loop in N iterations

						
							// Common variant
							for (var i = 0; i <= 5; i++) {
								// ...
							}
							// Another variant
							Array.from({length: 5}, (v, k) => {
								// ...
							});
							// Lodash
							_.times(5, (obj) => {
								// ...
							});

_.omit

Removing attributes from an object

						
							const user = { 
								name: 'John', 
								occupation: 'Architect', 
								street: 'Times Square Ave', 
								city: 'New York' 
							};
							// Common variant
							delete user.occupation;
							delete user.street;
							user; // { name: 'John', city: 'New York' }
							// Lodash
							_.omit(user, ['occupation', 'street']);

_.random

A pseudo-random number from a range

						
							// Common variant
							function getRandom(min, max) {
								return Math.floor(Math.random() * (max - min + 1)) + min;
							}
							getRandom(73, 1014);
							// Lodash
							_.random(73, 1014);

_.pluck

Ratio (map) for nested attributes

						
							const books = [
							{
								name: 'Pride and Prejudice',
								author: 'Jane Austen'
							},
							{
								name: 'Harry Potter and the Prisoner of Azkaban',
								author: 'J.K. Rowling'
							}];
							// Common variant
							books.map(book => book.name);
							// Lodash
							_.pluck(books, 'name');
